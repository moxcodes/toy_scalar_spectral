131 
132  // generate a very rough approximation for the first; working backward
133  // for Newton-Raphson nextAb < ab < prevAb
134  double prevAb = 1;
135  double ab= ((1 - 1/(double)pow(order,2) + 1/(double)pow(order,3))
136  * cos(PI*((double)(4 - 1))/((double)(4*order + 2))));
137  double nextAb;
138  for(int i = 1;i<=(order/2);i++)
139  {
140  nextAb= ((1 - 1/pow(order,2) + 1/pow(order,3))
141  * cos(PI*(4*(i+1) - 1)/(4*order + 2)));
142  legendre legendreN = legendre(order);

144  boost::math::tools::newton_raphson_iterate(legendreN,ab,

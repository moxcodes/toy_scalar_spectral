143  void operator() ( const std::vector<double> &x, std::vector<double> &dxdt, const double t)
144  {
145  // 2 steps : first evolve the bulk of each domain and get out the presumed time dependence
146  // at each interface, then impose agreement at each interface by averaging.
147  // This is slightly different from the Continuous Galerkin method, but only
148  // in the details of where the weight factors appear.
149 
150 
151  // bulk evolve and get boundary vals
152  std::vector<double> rightderivspi(doms+1);
153  std::vector<double> rightderivspsi(doms+1);
154  std::vector<double> leftderivspi(doms+1);
155  std::vector<double> leftderivspsi(doms+1);
156  //elstart tracks the position in the array of the start of the element, as they are of nonuniform size
157  std::vector<double> derivsCatcher;
158  int elstart = 0;
159  for(int i=0;i<doms;i++)
160  {
161  derivsCatcher = bulkEvolve(x,dxdt,i,elstart);
162  rightderivspi[i]=derivsCatcher[0];
163  rightderivspsi[i]=derivsCatcher[1];
164  leftderivspi[i+1]=derivsCatcher[2];
165  leftderivspsi[i+1]=derivsCatcher[3];
166  elstart+=2*n[i];
167  }
168  //leftmost and rightmost parts need to be modified to obey boundary conditions
169  leftderivspi[0] = -rightderivspi[0] + 2*boundData(t+1.0);
170  leftderivspsi[0] = rightderivspi[0];
171  rightderivspi[doms] = - leftderivspi[doms] - (reflect ? 0 : 2*leftderivspsi[doms]);
172  rightderivspsi[doms] = leftderivspsi[doms];
173  //edges evolve via averaging
174  elstart=0;
175  for(int i=0;i<doms;i++)
176  {
177  dxdt[elstart] = (leftderivspi[i] + rightderivspi[i])/2.0;
178  dxdt[elstart + n[i]] = (leftderivspsi[i] + rightderivspsi[i])/2.0;
179  dxdt[elstart + n[i] - 1] = (leftderivspi[i+1] + rightderivspi[i+1])/2.0;
180  dxdt[elstart + 2*n[i] - 1] = (leftderivspsi[i+1] + rightderivspsi[i+1])/2.0;
181  elstart+=2*n[i];
182  }
183  if((int)(t) == t && verbose)
184  printf("simulation time t=%f",t);
185  }

198  std::vector<double> bulkEvolve(const std::vector<double> &x, std::vector<double> &dxdt,int el,int elstart)
199  {
200  std::shared_ptr<std::vector<double>> pi(new std::vector<double>(x.begin()+elstart,x.begin()+elstart+n[el]));
201  std::shared_ptr<std::vector<double>> psi(new std::vector<double>(x.begin()+elstart+n[el],x.begin()+elstart+2*n[el]));
202  std::vector<double> dpsi = ((*DMats[el])*(*psi));
203  std::vector<double> dpi = ((*DMats[el])*(*pi));
204  for(int i=0;i<n[el];i++)
205  {
206  dxdt[elstart + i] = dpsi[i];
207  dxdt[elstart + n[el] + i] = dpi[i];
208  }
209  return std::vector<double>({dpsi[0],dpi[0],dpsi[n[el]-1],dpi[n[el]-1]});
210  }
